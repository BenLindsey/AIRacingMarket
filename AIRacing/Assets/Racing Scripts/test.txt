var maxSpeed = 15;

var dist = api.GetDistanceToCar(0);

if (dist > 0 && dist < 50) {
	maxSpeed = 25;
} 

if (api.CarInFront() || api.CarOnLeft()) {
	api.SteerToRight();
	maxSpeed = 25;
} else {
	api.SteerToLeft();
}

if (api.GetSpeed() > maxSpeed) { 
	api.SetThrottle(0f);
} else {
	api.SetThrottle(70f);
}
