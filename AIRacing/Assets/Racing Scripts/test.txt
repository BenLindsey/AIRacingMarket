var counter = "counter";
var speed = "speed";

var Init = function() {
	data[counter] = 0;
	data[speed] = 20;
};

var PhysicsUpdate = function() {

	data[counter] = data[counter] - 1;

	if (data[counter]  < 0) {
		if ((api.CarInFront() || api.CarOnLeft()) && !api.CarOnRight()) {
			data[counter] = 250;
			api.ChangeLaneRight();
			data[speed] = 25;
		} else {
			api.ChangeLaneLeft();
			data[speed] = 20;
		}
	}

	if (api.GetSpeed() > data[speed]) { 
		api.SetThrottle(0f);
	} else {
		api.SetThrottle(75f);
	}
};
